91  void HealthImpl::Run_handler(const NATIVE_INT_TYPE portNum, NATIVE_UINT_TYPE context) {
92  //dispatch messages
93  for (NATIVE_UINT_TYPE i = 0; i < this->queue_depth; i++) {
94  MsgDispatchStatus stat = this->doDispatch();
95  if (MSG_DISPATCH_EMPTY == stat) {
96  break;
97  }
98  FW_ASSERT(MSG_DISPATCH_OK == stat);
99  }
100 
101  if (this->m_enabled) {
102  // cycle through ping table, pinging ports that are not awaiting a reply
103  // for ports that are awaiting a reply, decrement their counters
104  // and check for violations
105 
106  for (NATIVE_UINT_TYPE entry = 0; entry < this->m_numPingEntries; entry++) {
107  if (HLTH_PING_ENABLED == this->m_pingTrackerEntries[entry].enabled) {
108  // If clear entry
109  if (0 == this->m_pingTrackerEntries[entry].cycleCount) {
110  // start a ping
111  this->m_pingTrackerEntries[entry].key = this->m_key;
112  // send ping
113  this->PingSend_out(entry, this->m_pingTrackerEntries[entry].key);
114  // increment key
115  this->m_key++;
116  // increment cycles for the entry
117  this->m_pingTrackerEntries[entry].cycleCount++;
118  } else {
119  // check to see if it is at warning threshold
120  if (this->m_pingTrackerEntries[entry].cycleCount ==
121  this->m_pingTrackerEntries[entry].entry.warnCycles) {
122  Fw::LogStringArg _arg = this->m_pingTrackerEntries[entry].entry.entryName;
123  this->log_WARNING_HI_HLTH_PING_WARN(_arg);
124  this->tlmWrite_PingLateWarnings(++this->m_warnings);
125  } else {
126  // check for FATAL timeout value
127  if (this->m_pingTrackerEntries[entry].entry.fatalCycles ==
128  this->m_pingTrackerEntries[entry].cycleCount) {
129  Fw::LogStringArg _arg = this->m_pingTrackerEntries[entry].entry.entryName;
130  this->log_FATAL_HLTH_PING_LATE(_arg);
131  }
132  } // if at warning or fatal threshold
133 
134  this->m_pingTrackerEntries[entry].cycleCount++;
135  } // if clear entry
136  } // if entry has ping enabled
137  } // for each entry
138 
139  // do other specialized platform checks (e.g. VxWorks suspended tasks)
140  this->doOtherChecks();
141 
142  } // If health checking is enabled
143 
144  // stroke watchdog.
145  if (this->isConnected_WdogStroke_OutputPort(0)) {
146  this->WdogStroke_out(0,this->m_watchDogCode);
147  }
148  }

166  // check to see if entry is in range
167  NATIVE_INT_TYPE entryIndex = this->findEntry(entry);
