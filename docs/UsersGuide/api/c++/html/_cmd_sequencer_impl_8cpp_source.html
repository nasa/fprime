153  void CmdSequencerComponentImpl::seqRunIn_handler(
154  NATIVE_INT_TYPE portNum,
155  Fw::EightyCharString &filename
156  ) {
157 
158  if (!this->requireRunMode(STOPPED)) {
159  this->seqDone_out(0,0,0,Fw::COMMAND_EXECUTION_ERROR);
160  return;
161  }
162 
163  // If file name is non-empty, load a file.
164  // Empty file name means don't load.
165  if (filename != "") {
166  Fw::CmdStringArg cmdStr(filename);
167  const bool status = this->loadFile(cmdStr);
168  if (!status) {
169  this->seqDone_out(0,0,0,Fw::COMMAND_EXECUTION_ERROR);
170  return;
171  }
172  }
173  else if (not this->m_sequence->hasMoreRecords()) {
174  // No sequence loaded
175  this->log_WARNING_LO_CS_NoSequenceActive();
176  this->error();
177  this->seqDone_out(0,0,0,Fw::COMMAND_EXECUTION_ERROR);
178  return;
179  }
180 
181  this->m_executedCount = 0;
182 
183  // Check the step mode. If it is auto, start the sequence
184  if (AUTO == this->m_stepMode) {
185  this->m_runMode = RUNNING;
186  this->performCmd_Step();
187  }
188 
189  this->log_ACTIVITY_HI_CS_PortSequenceStarted(this->m_sequence->getLogFileName());
190  }

240  void CmdSequencerComponentImpl ::
241  cmdResponseIn_handler(
242  NATIVE_INT_TYPE portNum,
243  FwOpcodeType opcode,
244  U32 cmdSeq,
245  Fw::CommandResponse response
246  )
247  {
248  if (this->m_runMode == STOPPED) {
249  // Sequencer is not running
250  this->log_WARNING_HI_CS_UnexpectedCompletion(opcode);
251  } else {
252  // clear command timeout
253  this->m_cmdTimeoutTimer.clear();
254  if (response != Fw::COMMAND_OK) {
255  this->commandError(this->m_executedCount, opcode, response);
256  this->performCmd_Cancel();
257  } else if (this->m_runMode == RUNNING && this->m_stepMode == AUTO) {
258  // Auto mode
259  this->commandComplete(opcode);
260  if (not this->m_sequence->hasMoreRecords()) {
261  // No data left
262  this->m_runMode = STOPPED;
263  this->sequenceComplete();
264  } else {
265  this->performCmd_Step();
266  }
267  } else {
268  // Manual step mode
269  this->commandComplete(opcode);
270  if (not this->m_sequence->hasMoreRecords()) {
271  this->m_runMode = STOPPED;
272  this->sequenceComplete();
273  }
274  }
275  }

280  {
281

389  void CmdSequencerComponentImpl::performCmd_Step(void) {
390 
391  this->m_sequence->nextRecord(m_record);
392  // set clock time base and context from value set when sequence was loaded
393  const Sequence::Header& header = this->m_sequence->getHeader();
394  this->m_record.m_timeTag.setTimeBase(header.m_timeBase);
395  this->m_record.m_timeTag.setTimeContext(header.m_timeContext);
396 
397  Fw::Time currentTime = this->getTime();
398  switch (this->m_record.m_descriptor) {
399  case Sequence::Record::END_OF_SEQUENCE:
400  this->m_runMode = STOPPED;
401  this->sequenceComplete();
402  break;
403  case Sequence::Record::RELATIVE:
404  this->performCmd_Step_RELATIVE(currentTime);
405  break;
406  case Sequence::Record::ABSOLUTE:
407  this->performCmd_Step_ABSOLUTE(currentTime);
408  break;
409  default:
410  FW_ASSERT(0, m_record.m_descriptor);
411  }
412  }
