270  ) const

274  char buffer[bufferSize];

277  buffer, sizeof(buffer), evalStr,

279  logFileName.toChar()
280  );
281  FW_ASSERT(static_cast<NATIVE_UINT_TYPE>(bytesCopied) < sizeof(buffer));
