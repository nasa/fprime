141  void UdpReceiverComponentImpl::doRecv(void) {
142 
143  // wait for data from the socket
144  NATIVE_INT_TYPE psize = recvfrom(
145  this->m_fd,
146  this->m_recvBuff.getBuffAddr(),
147  this->m_recvBuff.getBuffCapacity(),
148  MSG_WAITALL,
149  0,
150  0);
151  if (-1 == psize) {
152  if (errno != EINTR) {
153  Fw::LogStringArg arg(strerror(errno));
154  this->log_WARNING_HI_UR_RecvError(arg);
155  }
156  return;
157  }
158  // reset buffer for deserialization
159  Fw::SerializeStatus stat = this->m_recvBuff.setBuffLen(psize);
160  FW_ASSERT(Fw::FW_SERIALIZE_OK == stat, stat);
161 
162  // get sequence number
163  U8 seqNum;
164  stat = this->m_recvBuff.deserialize(seqNum);
165  // check for deserialization error or port number too high
166  if (stat != Fw::FW_SERIALIZE_OK) {
167  this->log_WARNING_HI_UR_DecodeError(DECODE_SEQ,stat);
168  this->m_decodeErrors++;
169  return;
170  }
171 
172  // track sequence number
173  if (this->m_firstSeq) {
174  // first time, set tracked sequence number equal to the one received
175  this->m_currSeq = seqNum;
176  this->m_firstSeq = false;
177  } else {
178  // make sure sequence number has gone up by one
179  if (seqNum != ++this->m_currSeq) {
180  // will only be right if it rolls over only once, but better than nothing
181  U8 diff = seqNum - this->m_currSeq;
182  this->m_packetsDropped += diff;
183  // send EVR
184  this->log_WARNING_HI_UR_DroppedPacket(diff);
185  // reset to current sequence
186  this->m_currSeq = seqNum;
187  }
188  }
189 
190  // get port number
191  U8 portNum;
192  stat = this->m_recvBuff.deserialize(portNum);
193  // check for deserialization error or port number too high
194  if (stat != Fw::FW_SERIALIZE_OK or portNum > this->getNum_PortsOut_OutputPorts()) {
195  this->log_WARNING_HI_UR_DecodeError(DECODE_PORT,stat);
196  this->m_decodeErrors++;
197  return;
198  }
199  // get buffer for port
200 
201  stat = this->m_recvBuff.deserialize(this->m_portBuff);
202  if (stat != Fw::FW_SERIALIZE_OK) {
203  this->log_WARNING_HI_UR_DecodeError(DECODE_BUFFER,stat);
204  this->m_decodeErrors++;
205  return;
206  }
207 
208  // call output port
209  DEBUG_PRINT("Calling port %d with %d bytes.\n",portNum,this->m_portBuff.getBuffLength());
210  if (this->isConnected_PortsOut_OutputPort(portNum)) {
211 
212  Fw::SerializeStatus stat = this->PortsOut_out(portNum,this->m_portBuff);
213 
214  // If had issues deserializing the data, then report it:
215  if (stat != Fw::FW_SERIALIZE_OK) {
216 
217  this->log_WARNING_HI_UR_DecodeError(PORT_SEND,stat);
218  this->m_decodeErrors++;
219  }
220 
221  this->m_packetsReceived++;
222  this->m_bytesReceived += psize;
223 
224  }
225  }
