102  void GroundInterfaceComponentImpl::frame_send(U8 *data, TOKEN_TYPE size, TOKEN_TYPE packet_type) {
103  // TODO: replace with a call to a buffer manager
104  Fw::Buffer buffer = m_ext_buffer;
105  Fw::ExternalSerializeBuffer buffer_wrapper(reinterpret_cast<U8*>(m_ext_buffer.getdata()),
106  m_ext_buffer.getsize());
107  // True size is supplied size plus sizeof(TOKEN_TYPE) if a packet_type other than "UNKNOWN" was supplied.
108  // This is because if not UNKOWN, the packet_type is serialized too. Otherwise it is assumed the PACKET_TYPE is
109  // already the first token in the UNKNOWN typed buffer.
110  U32 true_size = (packet_type != Fw::ComPacket::FW_PACKET_UNKNOWN) ? size + sizeof(TOKEN_TYPE) : size;
111  U32 total_size = sizeof(TOKEN_TYPE) + sizeof(TOKEN_TYPE) + true_size + sizeof(U32);
112  // Serialize data
113  FW_ASSERT(GND_BUFFER_SIZE >= total_size, GND_BUFFER_SIZE, total_size);
114  buffer_wrapper.serialize(START_WORD);
115  buffer_wrapper.serialize(static_cast<TOKEN_TYPE>(true_size));
116  // Explicitly set the packet type, if it didn't come with the data already
117  if (packet_type != Fw::ComPacket::FW_PACKET_UNKNOWN) {
118  buffer_wrapper.serialize(packet_type);
119  }
120  buffer_wrapper.serialize(data, size, true);
121  buffer_wrapper.serialize(static_cast<TOKEN_TYPE>(END_WORD));
122 
123  buffer.setsize(buffer_wrapper.getBuffLength());
124  buffer.setdata(reinterpret_cast<POINTER_CAST>(buffer_wrapper.getBuffAddr()));
125  FW_ASSERT(buffer.getsize() == total_size, buffer.getsize(), total_size);
126  write_out(0, buffer);
127  }

129  void GroundInterfaceComponentImpl ::
130  routeComData()
131  {
132  // Read the packet type from the data buffer
133  U32 packet_type = Fw::ComPacket::FW_PACKET_UNKNOWN;
134  m_in_ring.peek(packet_type, HEADER_SIZE);
135 
136  // Process variable type
137  switch (packet_type) {
138  case Fw::ComPacket::FW_PACKET_COMMAND: {
139  Fw::ComBuffer com;
140  m_in_ring.peek(com.getBuffAddr(), m_data_size, HEADER_SIZE);
141  // Reset com buffer for sending out data
142  com.setBuffLen(m_data_size);
143  uplinkPort_out(0, com, 0);
144  break;
145  }
146  case Fw::ComPacket::FW_PACKET_FILE: {
147  // If file uplink is possible, handle files. Otherwise ignore.
148  if (isConnected_fileUplinkBufferGet_OutputPort(0) &&
149  isConnected_fileDownlinkBufferSendOut_OutputPort(0)) {
150  Fw::Buffer buffer = fileUplinkBufferGet_out(0, m_data_size);
151  m_in_ring.peek(reinterpret_cast<U8*>(buffer.getdata()), m_data_size - sizeof(packet_type), HEADER_SIZE + sizeof(packet_type));
152  buffer.setsize(m_data_size - sizeof(packet_type));
153  fileUplinkBufferSendOut_out(0, buffer);
154  }
155  break;
156  }
157  default:
158  return;

162  void GroundInterfaceComponentImpl ::
163  processRing()
164  {
165  // Header items for the packet
166  TOKEN_TYPE start;
167  U32 checksum; //TODO: make this run a CRC32
168  // Inner-loop, process ring buffer looking for at least the header
169  while (m_in_ring.get_remaining_size() >= HEADER_SIZE) {
170  m_data_size = 0;
171  // Peek into the header and read out values
172  Fw::SerializeStatus status = m_in_ring.peek(start, 0);
173  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
174  status = m_in_ring.peek(m_data_size, sizeof(TOKEN_TYPE));
175  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
176  // Check the header for correctness
177  if (start != START_WORD || m_data_size >= MAX_DATA_SIZE) {
178  m_in_ring.rotate(1);
179  continue;
180  }
181  // Check for enough data to deserialize everything otherwise break and wait for more.
182  else if (m_in_ring.get_remaining_size() < (HEADER_SIZE + m_data_size + sizeof(END_WORD))) {
183  break;
184  }
185  // Continue with the data portion and checksum
186  m_in_ring.peek(checksum, HEADER_SIZE + m_data_size);
187  // Check checksum
188  if (checksum == END_WORD) {
189  routeComData();
190  m_in_ring.rotate(HEADER_SIZE + m_data_size + sizeof(U32));
191  }
192  // Failed checksum, keep looking for valid message
193  else {
194  m_in_ring.rotate(1);
195  }
196  }
197  }
