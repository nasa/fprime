112  void UdpSenderComponentImpl ::
113  PortsIn_handler(
114  NATIVE_INT_TYPE portNum,
115  Fw::SerializeBufferBase &Buffer
116  )
117  {
118  // return if we never successfully created the socket
119  if (-1 == this->m_fd) {
120  return;
121  }
122 
123  DEBUG_PRINT("PortsIn_handler: %d\n",portNum);
124  Fw::SerializeStatus stat;
125  m_sendBuff.resetSer();
126 
127  // serialize sequence number
128  stat = m_sendBuff.serialize(this->m_seq++);
129  FW_ASSERT(Fw::FW_SERIALIZE_OK == stat,stat);
130  // serialize port call
131  stat = m_sendBuff.serialize(static_cast<U8>(portNum));
132  FW_ASSERT(Fw::FW_SERIALIZE_OK == stat,stat);
133  // serialize port arguments buffer
134  stat = m_sendBuff.serialize(Buffer);
135  FW_ASSERT(Fw::FW_SERIALIZE_OK == stat,stat);
136  // send on UDP socket
137  DEBUG_PRINT("Sending %d bytes\n",m_sendBuff.getBuffLength());
138  ssize_t sendStat = sendto(this->m_fd,
139  m_sendBuff.getBuffAddr(),
140  m_sendBuff.getBuffLength(),
141  0,
142  (struct sockaddr *) &m_servAddr,
143  sizeof(m_servAddr));
144  if (-1 == sendStat) {
145  Fw::LogStringArg arg(strerror(errno));
146  this->log_WARNING_HI_US_SendError(arg);
147  } else {
148  FW_ASSERT((int)m_sendBuff.getBuffLength() == sendStat,(int)m_sendBuff.getBuffLength(),sendStat,portNum);
149  this->m_packetsSent++;
150  this->m_bytesSent += sendStat;
151  }
152  }
