36  void CommandDispatcherImpl::compCmdReg_handler(NATIVE_INT_TYPE portNum, FwOpcodeType opCode) {
37  // search for an empty slot
38  bool slotFound = false;
39  for (U32 slot = 0; slot < FW_NUM_ARRAY_ELEMENTS(this->m_entryTable); slot++) {
40  if ((not this->m_entryTable[slot].used) and (not slotFound)) {
41  this->m_entryTable[slot].opcode = opCode;
42  this->m_entryTable[slot].port = portNum;
43  this->m_entryTable[slot].used = true;
44  this->log_DIAGNOSTIC_OpCodeRegistered(opCode,portNum,slot);
45  slotFound = true;
46  } else if (this->m_entryTable[slot].used) { // make sure no duplicates
47  FW_ASSERT(this->m_entryTable[slot].opcode != opCode, opCode);
48  }
49  }
50  FW_ASSERT(slotFound,opCode);
51  }

53  void CommandDispatcherImpl::compCmdStat_handler(NATIVE_INT_TYPE portNum, FwOpcodeType opCode, U32 cmdSeq, Fw::CommandResponse response) {
54  // check response and log
55  if (Fw::COMMAND_OK == response) {
56  this->log_COMMAND_OpCodeCompleted(opCode);
57  } else {
58  this->m_numCmdErrors++;
59  this->tlmWrite_CommandErrors(this->m_numCmdErrors);
60  ErrorResponse evrResp = ERR_UNEXP;
61  switch (response) {
62  case Fw::COMMAND_INVALID_OPCODE:
63  evrResp = ERR_INVALID_OPCODE;
64  break;
65  case Fw::COMMAND_VALIDATION_ERROR:
66  evrResp = ERR_VALIDATION_ERROR;
67  break;
68  case Fw::COMMAND_FORMAT_ERROR:
69  evrResp = ERR_FORMAT_ERROR;
70  break;
71  case Fw::COMMAND_EXECUTION_ERROR:
72  evrResp = ERR_EXECUTION_ERROR;
73  break;
74  case Fw::COMMAND_BUSY:
75  evrResp = ERR_BUSY;
76  break;
77  case Fw::COMMAND_OK:
78  FW_ASSERT(0); // should never get here
79  break;
80  default:
81  evrResp = ERR_UNEXP;
82  break;
83  }
84  this->log_WARNING_HI_OpCodeError(opCode,evrResp);
85  }
86  // look for command source
87  NATIVE_INT_TYPE portToCall = -1;
88  U32 context;
89  for (U32 pending = 0; pending < FW_NUM_ARRAY_ELEMENTS(this->m_sequenceTracker); pending++) {
90  if (
91  (this->m_sequenceTracker[pending].seq == cmdSeq) &&
92  (this->m_sequenceTracker[pending].used)
93  ) {
94  portToCall = this->m_sequenceTracker[pending].callerPort;
95  context = this->m_sequenceTracker[pending].context;
96  FW_ASSERT(opCode == this->m_sequenceTracker[pending].opCode);
97  FW_ASSERT(portToCall < this->getNum_seqCmdStatus_OutputPorts());
98  this->m_sequenceTracker[pending].used = false;
99  break;
100  }
101  }
102 
103  if (portToCall != -1) {
104  // call port to report status
105  if (this->isConnected_seqCmdStatus_OutputPort(portToCall)) {
106  this->seqCmdStatus_out(portToCall,opCode,context,response);
107  }
108  }

111  void CommandDispatcherImpl::seqCmdBuff_handler(NATIVE_INT_TYPE portNum, Fw::ComBuffer &data, U32 context) {
112 
113  Fw::CmdPacket cmdPkt;
114  Fw::SerializeStatus stat = cmdPkt.deserialize(data);
115 
116  if (stat != Fw::FW_SERIALIZE_OK) {
117  CmdSerError serErr = ERR_UNEXP_STAT;
118  switch (stat) {
119  case Fw::FW_DESERIALIZE_BUFFER_EMPTY:
120  serErr = ERR_BUFFER_TOO_SMALL;
121  break;
122  case Fw::FW_DESERIALIZE_FORMAT_ERROR:
123  serErr = ERR_BUFFER_FORMAT;
124  break;
125  case Fw::FW_DESERIALIZE_SIZE_MISMATCH:
126  serErr = ERR_SIZE_MISMATCH;
127  break;
128  case Fw::FW_DESERIALIZE_TYPE_MISMATCH:
129  serErr = ERR_TYPE_MISMATCH;
130  break;
131  case Fw::FW_SERIALIZE_OK:
132  FW_ASSERT(0); // should never get here
133  break;
134  default:
135  serErr = ERR_UNEXP_STAT;
136  break;
137  }
138  this->log_WARNING_HI_MalformedCommand(serErr);
139  if (this->isConnected_seqCmdStatus_OutputPort(portNum)) {
140  this->seqCmdStatus_out(portNum,cmdPkt.getOpCode(),context,Fw::COMMAND_VALIDATION_ERROR);
141  }
142  return;
143  }
144 
145  // search for opcode in dispatch table
146  U32 entry;
147  bool entryFound = false;
148 
149  for (entry = 0; entry < FW_NUM_ARRAY_ELEMENTS(this->m_entryTable); entry++) {
150  if ((this->m_entryTable[entry].used) and (cmdPkt.getOpCode() == this->m_entryTable[entry].opcode)) {
151  entryFound = true;
152  break;
153  }
154  }
155  if (entryFound and this->isConnected_compCmdSend_OutputPort(this->m_entryTable[entry].port)) {
156  // register command in command tracker only if response port is connect
157  if (this->isConnected_seqCmdStatus_OutputPort(portNum)) {
158  bool pendingFound = false;
159 
160  for (U32 pending = 0; pending < FW_NUM_ARRAY_ELEMENTS(this->m_sequenceTracker); pending++) {
161  if (not this->m_sequenceTracker[pending].used) {
162  pendingFound = true;
163  this->m_sequenceTracker[pending].used = true;
164  this->m_sequenceTracker[pending].opCode = cmdPkt.getOpCode();
165  this->m_sequenceTracker[pending].seq = this->m_seq;
166  this->m_sequenceTracker[pending].context = context;
167  this->m_sequenceTracker[pending].callerPort = portNum;
168  break;
169  }
170  }
171 
172  // if we couldn't find a slot to track the command, quit
173  if (not pendingFound) {
174  this->log_WARNING_HI_TooManyCommands(cmdPkt.getOpCode());
175  if (this->isConnected_seqCmdStatus_OutputPort(portNum)) {
176  this->seqCmdStatus_out(portNum,cmdPkt.getOpCode(),context,Fw::COMMAND_EXECUTION_ERROR);
177  }
178  return;
179  }
180  } // end if status port connected
181  // pass arguments to argument buffer
182  this->compCmdSend_out(this->m_entryTable[entry].port,cmdPkt.getOpCode(),this->m_seq,cmdPkt.getArgBuffer());
183  // log dispatched command
184  this->log_COMMAND_OpCodeDispatched(cmdPkt.getOpCode(),this->m_entryTable[entry].port);
185 
186  // increment command count
187  this->m_numCmdsDispatched++;
188  // write telemetry channel for dispatched commands
189  this->tlmWrite_CommandsDispatched(this->m_numCmdsDispatched);
190  } else {
191  this->log_WARNING_HI_InvalidCommand(cmdPkt.getOpCode());
192  this->m_numCmdErrors++;
193  // Fail command back to port, if connected
194  if (this->isConnected_seqCmdStatus_OutputPort(portNum)) {
195  this->seqCmdStatus_out(portNum,cmdPkt.getOpCode(),context,Fw::COMMAND_INVALID_OPCODE);
196  }
197  this->tlmWrite_CommandErrors(this->m_numCmdErrors);
198  }
199 
200  // increment sequence number
201  this->m_seq++;
202  }
