458  void ActiveLoggerImpl::ALOG_SET_ID_FILTER_cmdHandler(
459  FwOpcodeType opCode,
460  U32 cmdSeq,
461  U32 ID,
462  IdFilterEnabled IdFilterEnable
463  ) {
464 
465  // check parameter
466  switch (IdFilterEnable) {
467  case ID_ENABLED:
468  case ID_DISABLED:
469  break;
470  default:
471  this->cmdResponse_out(opCode,cmdSeq,Fw::COMMAND_VALIDATION_ERROR);
472  return;
473  }
474 
475  if (ID_ENABLED == IdFilterEnable) { // add ID
476  // search list for existing entry
477  for (NATIVE_INT_TYPE entry = 0; entry < TELEM_ID_FILTER_SIZE; entry++) {
478  if (this->m_filteredIDs[entry] == ID) {
479  this->cmdResponse_out(opCode,cmdSeq,Fw::COMMAND_OK);
480  this->log_ACTIVITY_HI_ALOG_ID_FILTER_ENABLED(ID);
481  return;
482  }
483  }
484  // if not already a match, search for an open slot
485  for (NATIVE_INT_TYPE entry = 0; entry < TELEM_ID_FILTER_SIZE; entry++) {
486  if (this->m_filteredIDs[entry] == 0) {
487  this->m_filteredIDs[entry] = ID;
488  this->cmdResponse_out(opCode,cmdSeq,Fw::COMMAND_OK);
489  this->log_ACTIVITY_HI_ALOG_ID_FILTER_ENABLED(ID);
490  return;
491  }
492  }
493  // if an empty slot was not found, send an error event
494  this->log_WARNING_LO_ALOG_ID_FILTER_LIST_FULL(ID);
495  this->cmdResponse_out(opCode,cmdSeq,Fw::COMMAND_EXECUTION_ERROR);
496  } else { // remove ID
497  // search list for existing entry
498  for (NATIVE_INT_TYPE entry = 0; entry < TELEM_ID_FILTER_SIZE; entry++) {
499  if (this->m_filteredIDs[entry] == ID) {
500  this->m_filteredIDs[entry] = 0; // zero entry
501  this->cmdResponse_out(opCode,cmdSeq,Fw::COMMAND_OK);
502  this->log_ACTIVITY_HI_ALOG_ID_FILTER_REMOVED(ID);
503  return;
504  }
505  }
506  // if it gets here, wasn't found
507  this->log_WARNING_LO_ALOG_ID_FILTER_NOT_FOUND(ID);
508  this->cmdResponse_out(opCode,cmdSeq,Fw::COMMAND_EXECUTION_ERROR);
509  }
