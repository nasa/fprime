39  return SYSTEM_RESOURCES_ERROR;

42  if ((fp = fopen("/proc/stat", "r")) == nullptr) {
43 
44  return SYSTEM_RESOURCES_ERROR;
45 
46  }
47 
48  if (fgets(line, sizeof(line), fp) == nullptr) { //1st line. Aggregate cpu line.
49  fclose(fp);
50  return SYSTEM_RESOURCES_ERROR;
51  }
52 
53  for (U32 i = 0; i < cpu_index + 1; i++) {
54  if (fgets(line, sizeof(line), fp) == nullptr) { //cpu# line
55  fclose(fp);
56  return SYSTEM_RESOURCES_ERROR;
57  }
58  if (i != cpu_index) { continue; }
59 
60  if (!(line[0] == 'c' && line[1] == 'p' && line[2] == 'u')) {
61  fclose(fp);
62  return SYSTEM_RESOURCES_ERROR;
63  }
64  // No string concerns, as string is discarded
65  sscanf(line, "%*s %d %d %d %d", &cpu_data[0],
66  &cpu_data[1],
67  &cpu_data[2],
68  &cpu_data[3]); //cpu#: 4 numbers: usr, nice, sys, idle
69 
70  cpuUsed = cpu_data[0] + cpu_data[1] + cpu_data[2];
71  cpuTotal = cpu_data[0] + cpu_data[1] + cpu_data[2] + cpu_data[3];
72 
73  break;
74  }
75  fclose(fp);
76

78  cpu_ticks.total = cpuTotal;
