42  void ActiveTextLoggerComponentImpl::TextLogger_handler(NATIVE_INT_TYPE portNum,
43  FwEventIdType id,
44  Fw::Time &timeTag,
45  Fw::TextLogSeverity severity,
46  Fw::TextLogString &text)
47  {
48 
49  // Currently not doing any input filtering
50  // TKC - 5/3/2018 - remove diagnostic
51  if (Fw::TEXT_LOG_DIAGNOSTIC == severity) {
52  return;
53  }
54 
55  // Format the string here, so that it is done in the task context
56  // of the caller. Format doce borrowed from PassiveTextLogger.
57  const char *severityString = "UNKNOWN";
58  switch (severity) {
59  case Fw::TEXT_LOG_FATAL:
60  severityString = "FATAL";
61  break;
62  case Fw::TEXT_LOG_WARNING_HI:
63  severityString = "WARNING_HI";
64  break;
65  case Fw::TEXT_LOG_WARNING_LO:
66  severityString = "WARNING_LO";
67  break;
68  case Fw::TEXT_LOG_COMMAND:
69  severityString = "COMMAND";
70  break;
71  case Fw::TEXT_LOG_ACTIVITY_HI:
72  severityString = "ACTIVITY_HI";
73  break;
74  case Fw::TEXT_LOG_ACTIVITY_LO:
75  severityString = "ACTIVITY_LO";
76  break;
77  case Fw::TEXT_LOG_DIAGNOSTIC:
78  severityString = "DIAGNOSTIC";
79  break;
80  default:
81  severityString = "SEVERITY ERROR";
82  break;
83  }
84 
85  // TODO: Add calling task id to format string
86  char textStr[FW_INTERNAL_INTERFACE_STRING_MAX_SIZE];
87  NATIVE_INT_TYPE stat;
88 
89  if (timeTag.getTimeBase() == TB_WORKSTATION_TIME) {
90 
91  time_t t = timeTag.getSeconds();
92  // Using localtime_r prevents any other calls to localtime (from another thread for example) from
93  // interfering with our time object before we use it. However, the null pointer check is still needed
94  // to ensure a successful call
95  tm tm;
96  if (localtime_r(&t, &tm) == NULL) {
97  return;
98  }
99 
100  stat = snprintf(textStr,
101  FW_INTERNAL_INTERFACE_STRING_MAX_SIZE,
102  "EVENT: (%d) (%04d-%02d-%02dT%02d:%02d:%02d.%03u) %s: %s\n",
103  id, tm.tm_year + 1900, tm.tm_mon + 1, tm.tm_mday, tm.tm_hour,
104  tm.tm_min,tm.tm_sec,timeTag.getUSeconds(),
105  severityString,text.toChar());
106  }
107  else {
108 
109  stat = snprintf(textStr,
110  FW_INTERNAL_INTERFACE_STRING_MAX_SIZE,
111  "EVENT: (%d) (%d:%d,%d) %s: %s\n",
112  id,timeTag.getTimeBase(),timeTag.getSeconds(),timeTag.getUSeconds(),severityString,text.toChar());
113  }
114 
115  // If there was a error then just return:
116  if (stat <= 0) {
117  return;
118  }
119  // If there was string text truncation:
120  else if (stat >= FW_INTERNAL_INTERFACE_STRING_MAX_SIZE) {
121  // Do nothing
122  }
123 
124  // Call internal interface so that everything else is done on component thread,
125  // this helps ensure consistent ordering of the printed text:
126  Fw::InternalInterfaceString intText(textStr);
127  this->TextQueue_internalInterfaceInvoke(intText);
128  }
