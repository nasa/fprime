53  mathIn_handler(
54  const NATIVE_INT_TYPE portNum,
55  F32 val1,
56  F32 val2,
57  MathOperation operation
58  )
59  {
60  // declare result serializable
61  Ref::MathOp op;
62  F32 res = 0.0;
63  switch (operation) {
64  case MATH_ADD:
65  op.setop(ADD);
66  res = (val1 + val2)*this->m_factor1;
67  break;
68  case MATH_SUB:
69  op.setop(SUB);
70  res = (val1 - val2)*this->m_factor1;
71  break;
72  case MATH_MULTIPLY:
73  op.setop(MULT);
74  res = (val1 * val2)*this->m_factor1;
75  break;
76  case MATH_DIVIDE:
77  op.setop(DIVIDE);
78  res = (val1 / val2)*this->m_factor1;
79  break;
80  default:
81  FW_ASSERT(0,operation);
82  break;
83  }
84  Fw::ParamValid valid;
85  res = res/this->paramGet_factor2(valid);
86 
87  op.setval1(val1);
88  op.setval2(val2);
89  op.setresult(res);
90  this->log_ACTIVITY_HI_MR_OPERATION_PERFORMED(op);
91  this->tlmWrite_MR_OPERATION(op);
92  this->mathOut_out(0,res);
93  }
