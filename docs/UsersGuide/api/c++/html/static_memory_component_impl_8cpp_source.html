40 void StaticMemoryComponentImpl ::bufferDeallocate_handler(const NATIVE_INT_TYPE portNum, Fw::Buffer& fwBuffer) {
41  FW_ASSERT(static_cast<NATIVE_UINT_TYPE>(portNum) < FW_NUM_ARRAY_ELEMENTS(m_static_memory));
42  FW_ASSERT(m_allocated[portNum], portNum); // It is also an error to deallocate before returning
43  // Check the memory returned is within the region
44  FW_ASSERT(fwBuffer.getData() >= m_static_memory[portNum]);
45  FW_ASSERT((fwBuffer.getData() + fwBuffer.getSize()) <= (m_static_memory[portNum] + sizeof(m_static_memory[0])), fwBuffer.getSize(), sizeof(m_static_memory[0]));
46  m_allocated[portNum] = false;
47 }

49 Fw::Buffer StaticMemoryComponentImpl ::bufferAllocate_handler(const NATIVE_INT_TYPE portNum, U32 size) {
50  FW_ASSERT(static_cast<NATIVE_UINT_TYPE>(portNum) < FW_NUM_ARRAY_ELEMENTS(m_static_memory));
51  FW_ASSERT(size <= sizeof(m_static_memory[portNum])); // It is a topology error to ask for too much from this component
52  FW_ASSERT(not m_allocated[portNum], portNum); // It is also an error to allocate again before returning
53  m_allocated[portNum] = true;
54  Fw::Buffer buffer(m_static_memory[portNum], sizeof(m_static_memory[0]));
55  return buffer;
