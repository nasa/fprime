33 void DpWriter::bufferSendIn_handler(const NATIVE_INT_TYPE portNum, Fw::Buffer& buffer) {
34  Fw::Success::T status = Fw::Success::SUCCESS;
35  // portNum is unused
36  (void)portNum;
37  // Update num buffers received
38  ++this->m_numBuffersReceived;
39  // Check that the buffer is valid
40  if (!buffer.isValid()) {
41  this->log_WARNING_HI_InvalidBuffer();
42  status = Fw::Success::FAILURE;
43  }
44  // Check that the buffer is large enough to hold a data product packet
45  const FwSizeType bufferSize = buffer.getSize();
46  if (status == Fw::Success::SUCCESS) {
47  if (bufferSize < Fw::DpContainer::MIN_PACKET_SIZE) {
48  this->log_WARNING_HI_BufferTooSmallForPacket(
49  static_cast<U32>(bufferSize),
50  Fw::DpContainer::MIN_PACKET_SIZE);
51 
52  status = Fw::Success::FAILURE;
53  }
54  }
55  // Set up the container and check that the header hash is valid
56  Fw::DpContainer container;
57  if (status == Fw::Success::SUCCESS) {
58  container.setBuffer(buffer);
59  Utils::HashBuffer storedHash;
60  Utils::HashBuffer computedHash;
61  status = container.checkHeaderHash(storedHash, computedHash);
62  if (status != Fw::Success::SUCCESS) {
63  this->log_WARNING_HI_InvalidHeaderHash(
64  static_cast<U32>(bufferSize),
65  storedHash.asBigEndianU32(),
66  computedHash.asBigEndianU32());
67  }
68  }
69  // Deserialize the packet header
70  if (status == Fw::Success::SUCCESS) {
71  status = this->deserializePacketHeader(buffer, container);
72  }
73  // Check that the packet size fits in the buffer
74  if (status == Fw::Success::SUCCESS) {
75  const FwSizeType packetSize = container.getPacketSize();
76  if (bufferSize < packetSize) {
77  this->log_WARNING_HI_BufferTooSmallForData(
78  static_cast<U32>(bufferSize),
79  static_cast<U32>(packetSize));
80  status = Fw::Success::FAILURE;
81  }
82  }
83  // Perform the requested processing
84  if (status == Fw::Success::SUCCESS) {
85  this->performProcessing(container);
86  }
87  // Construct the file name
88  Fw::FileNameString fileName;
89  if (status == Fw::Success::SUCCESS) {
90  const FwDpIdType containerId = container.getId();
91  const Fw::Time timeTag = container.getTimeTag();
92  fileName.format(DP_FILENAME_FORMAT, this->m_dpFileNamePrefix.toChar(), containerId, timeTag.getSeconds(),
93  timeTag.getUSeconds());
94  }
95  FwSizeType fileSize = 0;
96  // Write the file
97  if (status == Fw::Success::SUCCESS) {
98  status = this->writeFile(container, fileName, fileSize);
99  }
100  // Send the DpWritten notification
101  if (status == Fw::Success::SUCCESS) {
102  this->sendNotification(container, fileName, fileSize);
103  }
104  // Deallocate the buffer
105  if (buffer.isValid()) {
106  this->deallocBufferSendOut_out(0, buffer);
107  }
108  // Update the error count
109  if (status != Fw::Success::SUCCESS) {
110  this->m_numErrors++;
111  }
112 }

161 void DpWriter::performProcessing(const Fw::DpContainer& container) {
162  // Get the buffer
163  Fw::Buffer buffer = container.getBuffer();
164  // Get the bit mask for the processing types
165  const Fw::DpCfg::ProcType::SerialType procTypes = container.getProcTypes();
166  // Do the processing
167  for (FwIndexType portNum = 0; portNum < NUM_PROCBUFFERSENDOUT_OUTPUT_PORTS; ++portNum) {
168  if ((procTypes & (1 << portNum)) != 0) {
169  this->procBufferSendOut_out(portNum, buffer);
170  }
171  }

174 Fw::Success::T DpWriter::writeFile(const Fw::DpContainer& container,
175  const Fw::FileNameString& fileName,
176  FwSizeType& fileSize) {
177  Fw::Success::T status = Fw::Success::SUCCESS;
178  // Get the buffer
179  Fw::Buffer buffer = container.getBuffer();
180  // Get the file size
181  fileSize = container.getPacketSize();
182  // Open the file
183  Os::File file;
184  Os::File::Status fileStatus = file.open(fileName.toChar(), Os::File::OPEN_CREATE);
185  if (fileStatus != Os::File::OP_OK) {
186  this->log_WARNING_HI_FileOpenError(static_cast<U32>(fileStatus), fileName);
187  status = Fw::Success::FAILURE;
188  }
189  // Write the file
190  if (status == Fw::Success::SUCCESS) {
191  // Set write size to file size
192  // On entry to the write call, this is the number of bytes to write
193  // On return from the write call, this is the number of bytes written
194  FwSignedSizeType writeSize = static_cast<FwSignedSizeType>(fileSize);
195  fileStatus = file.write(buffer.getData(), writeSize);
196  // If a successful write occurred, then update the number of bytes written
197  if (fileStatus == Os::File::OP_OK) {
198  this->m_numBytesWritten += static_cast<U64>(writeSize);
199  }
200  if ((fileStatus == Os::File::OP_OK) and (writeSize == static_cast<FwSignedSizeType>(fileSize))) {
201  // If the write status is success, and the number of bytes written
202  // is the expected number, then record the success
203  this->log_ACTIVITY_LO_FileWritten(
204  static_cast<U32>(writeSize),
205  fileName);
206  } else {
207  // Otherwise record the failure
208  this->log_WARNING_HI_FileWriteError(static_cast<U32>(fileStatus), static_cast<U32>(writeSize),
209  static_cast<U32>(fileSize), fileName);
210  status = Fw::Success::FAILURE;
211  }
212  }
213  // Update the count of successful or failed writes
214  if (status == Fw::Success::SUCCESS) {
215  this->m_numSuccessfulWrites++;
216  } else {
217  this->m_numFailedWrites++;
218  }
219  // Return the status
220  return status;
221 }
