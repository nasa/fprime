94 void TcpServerComponentImpl::readLoop() {
95  Drv::SocketIpStatus status = Drv::SocketIpStatus::SOCK_NOT_STARTED;
96  // Keep trying to reconnect until the status is good, told to stop, or reconnection is turned off
97  do {
98  status = this->startup();
99  if (status != SOCK_SUCCESS) {
100  Fw::Logger::log("[WARNING] Failed to listen on port %hu with status %d\n", this->getListenPort(), status);
101  (void)Os::Task::delay(SOCKET_RETRY_INTERVAL);
102  continue;
103  }
104  }
105  while (this->running() && status != SOCK_SUCCESS && this->m_reconnect);
106  // If start up was successful then perform normal operations
107  if (this->running() && status == SOCK_SUCCESS) {
108  // Perform the nominal read loop
109  SocketComponentHelper::readLoop();
110  // Terminate the server
111  this->terminate();
112  }
113 }
