68  void SocketIpDriverComponentImpl::readTask(void* pointer) {
69  FW_ASSERT(pointer);
70  SocketIpStatus status = SOCK_SUCCESS;
71  SocketIpDriverComponentImpl* self = reinterpret_cast<SocketIpDriverComponentImpl*>(pointer);
72  do {
73  // Open a network connection if it has not already been open
74  if (not self->m_helper.isOpened()
75  and (self->m_helper.open() != SOCK_SUCCESS)
76  and not self->m_stop) {
77  Os::Task::delay(PRE_CONNECTION_RETRY_INTERVAL_MS);
78  }
79 
80  // If the network connection is open, read from it
81  if (self->m_helper.isOpened()) {
82  BYTE* data = reinterpret_cast<U8*>(self->m_buffer.getdata());
83  FW_ASSERT(data);
84  I32 size = 0;
85  status = self->m_helper.recv(data,size);
86  if (status != SOCK_SUCCESS &&
87  status != SOCK_INTERRUPTED_TRY_AGAIN) {
88  self->m_helper.close();
89  } else {
90  // Ignore KEEPALIVE data and send out any other data.
91  if (memcmp(data, KEEPALIVE_CONST,
92  (size > static_cast<I32>(sizeof(KEEPALIVE_CONST)) - 1) ? sizeof(KEEPALIVE_CONST) -1 : size) != 0) {
93  self->m_buffer.setsize(size);
94  self->recv_out(0, self->m_buffer);
95  }
96  }
97  }
98  }
99  // As long as not told to stop, and we are successful interrupted or ordered to retry, keep receiving
100  while(not self->m_stop && (status == SOCK_SUCCESS ||
101  status == SOCK_INTERRUPTED_TRY_AGAIN || RECONNECT_AUTOMATICALLY != 0));
102  }
