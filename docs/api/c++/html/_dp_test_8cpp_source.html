106void DpTest ::dpRecv_Container1_handler(DpContainer& container, Fw::Success::T status) {
107 // Test container assignment
108 this->m_container = container;
109 if (status == Fw::Success::SUCCESS) {
110 auto serializeStatus = Fw::FW_SERIALIZE_OK;
111 for (FwSizeType i = 0; i < CONTAINER_1_DATA_SIZE; ++i) {
112 serializeStatus = this->m_container.serializeRecord_U32Record(this->u32RecordData);
113 if (serializeStatus == Fw::FW_SERIALIZE_NO_ROOM_LEFT) {
114 break;
115 }
116 FW_ASSERT(serializeStatus == Fw::FW_SERIALIZE_OK, status);
117 }
118 // Use the time stamp from the time get port
119 this->dpSend(this->m_container);
120 // Check that buffer is no longer valid
121 FW_ASSERT(!this->m_container.getBuffer().isValid());
122 }
123}
