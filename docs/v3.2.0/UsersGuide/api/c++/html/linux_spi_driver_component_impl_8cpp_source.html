36 void LinuxSpiDriverComponentImpl::SpiReadWrite_handler(
37 const NATIVE_INT_TYPE portNum, Fw::Buffer &writeBuffer,
38 Fw::Buffer &readBuffer) {
39
40 if (this->m_fd == -1) {
41 return;
42 }
43
44 DEBUG_PRINT("Writing %d bytes to SPI\n",writeBuffer.getSize());
45
46 spi_ioc_transfer tr;
47 // Zero for unused fields:
48 memset(&tr, 0, sizeof(tr));
49 tr.tx_buf = reinterpret_cast<__u64>(writeBuffer.getData());
50 tr.rx_buf = reinterpret_cast<__u64>(readBuffer.getData());
51 tr.len = writeBuffer.getSize();
52/*
53 .speed_hz = 0,
54 .delay_usecs = 0,
55 .bits_per_word = 0,
56 .cs_change = 0,
57 .tx_nbits = 0, // on more-recent kernel versions;
58 .rx_nbits = 0, // on more-recent kernel versions;
59 .pad = 0
60*/
61
62 NATIVE_INT_TYPE stat = ioctl(this->m_fd, SPI_IOC_MESSAGE(1), &tr);
63
64 if (stat < 1) {
65 this->log_WARNING_HI_SPI_WriteError(this->m_device,this->m_select,stat);
66 }
67 this->m_bytes += readBuffer.getSize();
68 this->tlmWrite_SPI_Bytes(this->m_bytes);
69 }

71 bool LinuxSpiDriverComponentImpl::open(NATIVE_INT_TYPE device,
72 NATIVE_INT_TYPE select,
73 SpiFrequency clock) {
74
75 this->m_device = device;
76 this->m_select = select;
77 NATIVE_INT_TYPE fd;
78 NATIVE_INT_TYPE ret;
79
80 // Open:
81 char devName[256];
82 snprintf(devName,sizeof(devName),"/dev/spidev%d.%d",device,select);
83 // null terminate
84 devName[sizeof(devName)-1] = 0;
85 DEBUG_PRINT("Opening SPI device %s\n",devName);
86
87 fd = ::open(devName, O_RDWR);
88 if (fd == -1) {
89 DEBUG_PRINT("open SPI device %d.%d failed. %d\n",device,select,errno);
90 this->log_WARNING_HI_SPI_OpenError(device,select,fd);
91 return false;
92 } else {
93 DEBUG_PRINT("Successfully opened SPI device %s fd %d\n",devName,fd);
94 }
95
96 this->m_fd = fd;
97
98 // Configure:
99 /*
100 * SPI Mode 0
101 */
102 U8 mode = SPI_MODE_0; // Mode 0 (CPOL = 0, CPHA = 0)
103 ret = ioctl(fd, SPI_IOC_WR_MODE, &mode);
104 if (ret == -1) {
105 DEBUG_PRINT("ioctl SPI_IOC_WR_MODE fd %d failed. %d\n",fd,errno);
106 this->log_WARNING_HI_SPI_ConfigError(device,select,ret);
107 return false;
108 } else {
109 DEBUG_PRINT("SPI fd %d WR mode successfully configured to %d\n",fd,mode);
110 }
111
112 ret = ioctl(fd, SPI_IOC_RD_MODE, &mode);
113 if (ret == -1) {
114 DEBUG_PRINT("ioctl SPI_IOC_RD_MODE fd %d failed. %d\n",fd,errno);
115 this->log_WARNING_HI_SPI_ConfigError(device,select,ret);
116 return false;
117 } else {
118 DEBUG_PRINT("SPI fd %d RD mode successfully configured to %d\n",fd,mode);
119 }
120
121 /*
122 * 8 bits per word
123 */
124 U8 bits = 8;
125 ret = ioctl(fd, SPI_IOC_WR_BITS_PER_WORD, &bits);
126 if (ret == -1) {
127 DEBUG_PRINT("ioctl SPI_IOC_WR_BITS_PER_WORD fd %d failed. %d\n",fd,errno);
128 this->log_WARNING_HI_SPI_ConfigError(device,select,ret);
129 return false;
130 } else {
131 DEBUG_PRINT("SPI fd %d WR bits per word successfully configured to %d\n",fd,bits);
132 }
133
134 ret = ioctl(fd, SPI_IOC_RD_BITS_PER_WORD, &bits);
135 if (ret == -1) {
136 DEBUG_PRINT("ioctl SPI_IOC_RD_BITS_PER_WORD fd %d failed. %d\n",fd,errno);
137 this->log_WARNING_HI_SPI_ConfigError(device,select,ret);
138 return false;
139 } else {
140 DEBUG_PRINT("SPI fd %d RD bits per word successfully configured to %d\n",fd,bits);
141 }
142
143 /*
144 * Max speed in Hz
145 */
146 ret = ioctl(fd, SPI_IOC_WR_MAX_SPEED_HZ, &clock);
147 if (ret == -1) {
148 DEBUG_PRINT("ioctl SPI_IOC_WR_MAX_SPEED_HZ fd %d failed. %d\n",fd,errno);
149 this->log_WARNING_HI_SPI_ConfigError(device,select,ret);
150 return false;
151 } else {
152 DEBUG_PRINT("SPI fd %d WR freq successfully configured to %d\n",fd,clock);
153 }
154
155 ret = ioctl(fd, SPI_IOC_RD_MAX_SPEED_HZ, &clock);
156 if (ret == -1) {
157 DEBUG_PRINT("ioctl SPI_IOC_RD_MAX_SPEED_HZ fd %d failed. %d\n",fd,errno);
158 this->log_WARNING_HI_SPI_ConfigError(device,select,ret);
159 return false;
160 } else {
161 DEBUG_PRINT("SPI fd %d RD freq successfully configured to %d\n",fd,clock);
162 }
163
164 return true;
165
166 }
