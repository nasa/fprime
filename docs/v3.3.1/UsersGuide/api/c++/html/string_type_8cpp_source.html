110 void StringBase::appendBuff(const CHAR* buff, NATIVE_UINT_TYPE size) {
111 const U32 capacity = this->getCapacity();
112 const U32 length = this->length();
113 FW_ASSERT(capacity > length, capacity, length);
114 // Subtract 1 to leave space for null terminator
115 U32 remaining = capacity - length - 1;
116 if(size < remaining) {
117 remaining = size;
118 }
119 FW_ASSERT(remaining < capacity, remaining, capacity);
120 (void) strncat(const_cast<CHAR*>(this->toChar()), buff, remaining);
121 }
